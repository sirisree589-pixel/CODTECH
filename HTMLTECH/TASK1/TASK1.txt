    <div id="quiz">
      <div class="question" id="qText">Loading question...</div>
      <div class="answers" id="answers"></div>

        <div class="score" id="score">Score: 0</div>

        </div>

      <div class="footer">
        <small id="progress">0 / 0</small>
      </div>

  <script>
    // ====== Quiz Data (can be loaded dynamically) ======
    const questions = [
      {
        id: 1,
        q: "Which HTML element do we use for the largest heading?",
        options: ["<head>", "<h1>", "<heading>", "<title>"],
        answerIndex: 1
      },
      {
        id: 2,
        q: "What is the output of: console.log(typeof null) ?",
        options: ["'null'", "'object'", "'undefined'", "'number'"],
        answerIndex: 1
      },
      {
        id: 3,
        q: "Which CSS property controls spacing inside an element?",
        options: ["margin", "padding", "border-spacing", "gap"],
        answerIndex: 1
      },
      {
        id: 4,
        q: "Which tag is used for an unordered list in HTML?",
        options: ["<ol>", "<ul>", "<li>", "<list>"],
        answerIndex: 1
      },
      {
        id: 5,
        q: "Which operator is used to assign a value in JavaScript?",
        options: ["==", "===", "=", ":="],
        answerIndex: 2
      }
    ];

    // ======= App State =======
    let currentIndex = 0;
    let score = 0;
    let answered = false;

    const qText = document.getElementById('qText');
    const answersDiv = document.getElementById('answers');
    const scoreEl = document.getElementById('score');
    const progress = document.getElementById('progress');
    const nextBtn = document.getElementById('nextBtn');

    function loadQuestion(index){
      answered = false;
      const item = questions[index];
      qText.textContent = item.q;
      answersDiv.innerHTML = '';
      item.options.forEach((opt, i) => {
        const b = document.createElement('button');
        b.className = 'answer';
        b.innerHTML = opt;
        b.dataset.index = i;
        b.onclick = onAnswerClick;
        answersDiv.appendChild(b);
      });
      progress.textContent = `${index+1} / ${questions.length}`;
    }

    function onAnswerClick(e){
      if(answered) return;
      answered = true;
      const chosen = Number(e.currentTarget.dataset.index);
      const correct = questions[currentIndex].answerIndex;
      const buttons = answersDiv.querySelectorAll('.answer');
      buttons.forEach(btn => {
        const idx = Number(btn.dataset.index);
        if(idx === correct) btn.classList.add('correct');
        if(idx === chosen && idx !== correct) btn.classList.add('wrong');
        // disable pointer
        btn.style.pointerEvents = 'none';
      });

      if(chosen === correct){
        score += 10;
      }
      scoreEl.textContent = `Score: ${score}`;
    }

    nextBtn.addEventListener('click', () => {
      if(!answered){
        // allow skipping but mark none selected
      }
      currentIndex++;
      if(currentIndex >= questions.length){
        showResults();
        return;
      }
      loadQuestion(currentIndex);
    });

    function showResults(){
      answersDiv.innerHTML = '';
      qText.innerHTML = `Quiz Complete! Final Score: ${score}`;
      const restart = document.createElement('button');
      restart.textContent = 'Restart Quiz';
      restart.onclick = () => { currentIndex = 0; score = 0; scoreEl.textContent = 'Score: 0'; loadQuestion(0); };
      answersDiv.appendChild(restart);
      progress.textContent = `${questions.length} / ${questions.length}`;
    }

    // Init
    loadQuestion(0);
  </script>
